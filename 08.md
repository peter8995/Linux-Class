# Line notify
* 到line notify為指定群組產生token
* #!/bin/bash
  \n# LINE Notify Token - Media > "Send to".
  TOKEN=""

  \# {ALERT.SUBJECT}
  \subject="$1"

  \# {ALERT.MESSAGE}
  message="$2"
* chmod +x line.sh
* ./line.sh test hello
  * 傳送hello到指定群組

curl https://notify-api.line.me/api/notify -H "Authorization: Bearer ${TOKEN}" -d "message=${message}"
